App Structure

The file has a .sqc extension.

It must contain a: main() function.

#Step
0Include SQL support
1Declare
2Connect to DB
3Do work
4Process Errors
5Commit/Abort
6Disconnect

Syntax

An SQL statement can be inserted by embedding the following:

EXEC SQL <SQL statement>;
<SQL statement>?Notes
BEGIN DECLARE SECTION;
(declare host variables)
END DECLARE SECTION;
Declare host variables in a declare section:
Then they can be used in
EXEC SQL statements by prefixing with a :
INCLUDE SQLCASQLCA - SQL communication area

Handling Null Values

NULL is not a valid result

Without the indicator variable, NULL results in SQLERROR

Indicator Variable

If ind < 0 → NULL 

smallint ind; // (within a BEGIN/END DECLARE SECTION)

	       FROM ... ;

Cursor

For queries returning more than 1 answer

  1. Declaring a cursor
EXEC SQL DECLARE <name> CURSOR FOR <query>
  1. Iterator protocol
EXEC SQL OPEN <name>;
//------------------------------------------
EXEC SQL WHENEVER NOT FOUND GO TO end; 
for(;;) {
	//set up host params: 
	EXEC SQL FETCH <name> INTO <host params>; 
	//process the fetched tuple 
};
end:
//------------------------------------------
EXEC SQL CLOSE <name>;

  • EX 2

    In this example, we will execute simple DML statements (queries with a single tuple, constant statements, simple parameters)

    #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "util.h"

//----------------------INCLUDE----------------------

EXEC SQL INCLUDE SQLCA;

int main(int argc, char *argv[]) {

	//----------------------DECLARE----------------------
	int i;
	
	EXEC SQL BEGIN DECLARE SECTION;
	  char db[6] = "cs348";
	  char title[72], pubid[9];
	EXEC SQL END DECLARE SECTION;

	//----------------------CONNECT----------------------
	
	EXEC SQL CONNECT TO :db;
	
  //----------------------ERROR HANDLE----------------------
	
	EXEC SQL WHENEVER SQLERROR  GO TO error;

	//----------------------DO WORK----------------------
	
	for (i=1; i<argc; i++) {

		strncpy(pubid,argv[i],8); //get a pubid from main arguments
		
		EXEC SQL WHENEVER NOT FOUND GO TO nope;

		EXEC SQL SELECT title INTO :title
		         FROM   publication
		         WHERE  pubid = :pubid;
		
		continue;
		nope:
			printf("*** not found ***", )
	};

	//----------------------COMMIT----------------------
	EXEC SQL COMMIT;
	//----------------------DISCONNECT----------------------
	EXEC SQL CONNECT reset;
	exit(0);
	//----------------------ERROR HANDLE----------------------	

	error:
		check_error("My error",&sqlca);
		EXEC SQL WHENEVER SQLERROR CONTINUE;
		EXEC SQL ROLLBACK;
		EXEC SQL CONNECT reset;
		exit(1);
}

    #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "util.h"

//----------------------INCLUDE----------------------

EXEC SQL INCLUDE SQLCA; 

//----------------------DECLARE----------------------

EXEC SQL BEGIN DECLARE SECTION;
   char db[6] = "cs348";
   char title[72], name[20], apat[10];
   short aid;
EXEC SQL END DECLARE SECTION;

int main(int argc, char *argv[]) {

   if (argc!=2) {
      printf("Usage: sample3 <pattern>\n");
      exit(1);
   };

   EXEC SQL WHENEVER SQLERROR  GO TO error;

	 //----------------------CONNECT----------------------

   EXEC SQL CONNECT TO :db;

	 //----------------------DO WORK----------------------

   strncpy(apat,argv[1],8); // arg 1 -> apat
     
   EXEC SQL DECLARE author CURSOR FOR 
             SELECT name, title 
             FROM author , wrote, publication
             WHERE name LIKE :apat
               AND aid=author
               AND pubid=publication;

   EXEC SQL OPEN author;

   EXEC SQL WHENEVER NOT FOUND GO TO end;
   for (;;) {
     EXEC SQL FETCH author INTO :name, :title;
     printf("%10s -> %20s: %s\n",apat,name,title);
   };

end:

   EXEC SQL CLOSE author;
   EXEC SQL COMMIT;
   EXEC SQL CONNECT RESET;
   exit(0);

error:
   check_error("My error",&sqlca);
   EXEC SQL WHENEVER SQLERROR CONTINUE;
   EXEC SQL ROLLBACK;
   EXEC SQL CONNECT reset;
   exit(1);
}

Stored Procedures

Syntax

CREATE FUNCTION <Name>(<ParamName> <ParamType>)
	RETURNS <ReturnType>
RETURN
	<UnaryQuery>

Examples

  • Ex 1
    CREATE FUNCTION sumSalaries(dept CHAR(3))
	RETURNS DECIMAL(9,2)
RETURN
	SELECT sum(salary)
	FROM employee
	WHERE workdept = dept

    Using it in DB2:

    db2 => SELECT deptno, sumSalaries(depptno) AS sal FROM department
  • Ex 2
    CREATE FUNCTION deptSalaries(dept, char(3))
	RETURNS TABLE(salary DECIMAL(9,2))
RETURN 
	SELECT salary
	FROM employee
	WHERE workdept = dept

    Using it:

    SELECT * 
FROM TABLE (deptSalaries(CAST('A00' AS CHAR(3)))) AS s
