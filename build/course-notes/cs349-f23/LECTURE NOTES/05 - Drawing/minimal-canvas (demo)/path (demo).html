function pathDemo(gc: CanvasRenderingContext2D)
{
  gc.lineWidth = 3;

  // line
  gc.strokeStyle = "black";
  gc.beginPath();
  gc.moveTo(10, 10);
  gc.lineTo(60, 60);
  gc.stroke();

  // polyline or polygon
  gc.strokeStyle = "green";
  gc.beginPath();
  gc.moveTo(100, 10);
  gc.lineTo(150, 60);
  gc.lineTo(100, 60);
  gc.closePath();      // line back to first point
  gc.stroke();

  // BLUE circle (using ellipse)
  gc.strokeStyle = "blue";
  gc.beginPath();
  gc.ellipse(200, 30, 25, 25, 0, 0, 2 * Math.PI);
  gc.stroke();

  // RED circle (using arc)
  gc.strokeStyle = "red";
  gc.beginPath();
  gc.arc(200, 30, 20, 0, 2 * Math.PI);
  gc.stroke();
}
