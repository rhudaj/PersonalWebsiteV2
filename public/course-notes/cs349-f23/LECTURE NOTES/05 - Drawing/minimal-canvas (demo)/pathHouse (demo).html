function pathHouseDemo(gc: CanvasRenderingContext2D)
{
  // array
  const housePoints = [
    [40, 0],
    [80, 40],
    [80, 100],
    [0, 100],
    [0, 40],
  ];

  gc.lineWidth = 3;
  gc.strokeStyle = "red";

  gc.beginPath();               //don't need moveTo() after this
  housePoints.forEach((p) => {
    const [x, y] = p;           //destructing p
    gc.lineTo(x, y);
  });

  gc.closePath();
  gc.stroke();
}

function colourDemo(gc: CanvasRenderingContext2D) {
  for (let i = 0; i < 10; i++) {
    const h = Math.random() * 360;
    gc.fillStyle = `hsl(${h}deg 80% 50%)`;
    gc.fillRect(i * 20, 20, 20, 20);
  }
}
