  return func() + name;
}

  1. Return Function
function makeGreeting(msg) {
  return function(name) { return msg + name; };
}
const greeting = makeGreeting("Hi, ");  //variable <- function
const greeting2 = greeting("Bob");    //call variable
console.log(greeting2);  // "Hi, bob"

Method 1

function Shape(colour) {
  this.colour = colour;
  this.draw = function() {...};
}



function Square(colour, size) {
  Shape.call(this, colour);
  this.size = size;
  this.draw = function() {...}
}

Method 2

class Shape {
  constructor(colour) {
    this.colour = colour;
  }
  draw = function() {...}
}

class Square {
  constructor(colour, size) {
    super(colour);
    this.size = size;
  }
  draw = function() {...}
}

Union Types

Use multiple types with |

	console.log(`Your ID is: ${id}`);
}

Type Guarding

const el = document.getElementById("apple");

if (el) el.setAttribute("class", "red");

Needed for union types

function formatId(id: number | string): string {
	if (typeof id === "number") {
		return `${id}`
	}
	else {
		return id.toUpperCase();
	}
}
